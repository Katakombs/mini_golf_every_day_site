# IMPORTANT: Before setting up cron, update the paths in cron_wrapper.sh:

# - PYTHON_BIN="/home/phazeshi/virtualenv/minigolfeveryday/3.8/bin/python3.8"  (your Python path)

# To install this cron job:
# 1. Edit your cron jobs: crontab -e
# 2. Add one of the lines below (uncomment the one you want)
# 3. Save and exit

# To view current cron jobs: crontab -l
# To remove all cron jobs: crontab -r

# Option 1: Check for new videos every 2 hours (recommended for daily content)
# This will run at: 12:00 AM, 2:00 AM, 4:00 AM, 6:00 AM, 8:00 AM, 10:00 AM, 12:00 PM, 2:00 PM, 4:00 PM, 6:00 PM, 8:00 PM, 10:00 PM
0 */2 * * * /home/phazeshi/minigolfeveryday/cron_wrapper.sh >> /home/phazeshi/minigolfeveryday/cron.log 2>&1

# Option 2: Check for new videos every 4 hours (more conservative)
# This will run at: 12:00 AM, 4:00 AM, 8:00 AM, 12:00 PM, 4:00 PM, 8:00 PM
# 0 */4 * * * /home/phazeshi/minigolfeveryday/cron_wrapper.sh >> /home/phazeshi/minigolfeveryday/cron.log 2>&1

# Option 3: Check for new videos twice daily (morning and evening)
# This will run at: 8:00 AM and 8:00 PM
# 0 8,20 * * * /home/phazeshi/minigolfeveryday/cron_wrapper.sh >> /home/phazeshi/minigolfeveryday/cron.log 2>&1

# Option 4: Check for new videos once daily at 9:00 AM
# 0 9 * * * /home/phazeshi/minigolfeveryday/cron_wrapper.sh >> /home/phazeshi/minigolfeveryday/cron.log 2>&1

# Option 5: Check for new videos every hour during posting hours (8 AM to 10 PM)
# 0 8-22 * * * /home/phazeshi/minigolfeveryday/cron_wrapper.sh >> /home/phazeshi/minigolfeveryday/cron.log 2>&1

# Note: All times are in server timezone. Adjust as needed for your timezone.
# The >> redirects output to cron.log, and 2>&1 redirects errors to the same file.
